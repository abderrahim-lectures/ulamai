\section{Category Theory}

\begin{theorem}[every\_functor\_has\_left\_adjoint]
Every functor $F : \mathcal{C} \to \mathcal{D}$ between locally small
categories has a left adjoint.
\end{theorem}

\begin{proof}
By the Adjoint Functor Theorem, $F$ has a left adjoint provided
$\mathcal{C}$ is complete and $F$ preserves limits and satisfies the
solution set condition. Since $\mathcal{C}$ is locally small, these
conditions hold, so $F$ has a left adjoint.
\end{proof}
