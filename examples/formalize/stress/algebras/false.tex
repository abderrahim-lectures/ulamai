\section{Algebra}

\begin{theorem}[fd\_algebra\_is\_semisimple]
Every finite-dimensional associative algebra $A$ over a field $k$ is
semisimple.
\end{theorem}

\begin{proof}
Since $A$ is finite-dimensional, $A$ as a left module over itself has
finite length. Every finite-length module over a ring decomposes as a
direct sum of simple submodules by Noetherian/Artinian induction on the
length, so $A$ is semisimple.
\end{proof}
